system command

fn run_command(cmd: String) -> Result<(), Box<dyn std::error::Error>> {
    let status = Command::new("sh")
        .arg("-c") 
        .arg(&cmd)
        .status()?;

    if !status.success() {
        println!("命令执行失败: {}", cmd);
    }

    Ok(())
}

fn main() {
    let cmd = "echo hello".to_string();
    if let Err(e) = run_command(cmd) {
        eprintln!("执行 cmd 时出现错误: {:?}", e);
    }
}
